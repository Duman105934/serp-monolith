
# ==============================
#   ШАБЛОН СООБЩЕНИЯ КОММИТА
# ==============================
# Формат:
# <type>(<scope>): <описание>
#
# Примеры:
# feat(auth): добавлен экран входа и регистрации
# fix(api): исправлена обработка ошибок при логине
# refactor(worker): вынес логику отправки уведомлений
# chore(deps): обновлены зависимости shadcn и nestjs
#
# ---------------------------------
# Доступные типы коммитов
# ---------------------------------
# feat       — добавление новой функциональности
# fix        — исправление ошибки
# refactor   — переработка без изменения поведения
# style      — форматирование, правки CSS, шрифтов
# docs       — документация, README, Swagger
# chore      — мелкие задачи, конфиги, зависимости
# build      — изменения сборки, Docker, CI/CD
# perf       — оптимизация производительности
# test       — добавление/изменение тестов
# ci         — настройка или исправление CI/CD pipeline
# revert     — откат предыдущего коммита
#
# ---------------------------------
# Области (scope) — указываются по желанию
# ---------------------------------
# ui, auth, api, db, worker, gateway, analytics,
# camunda, reports, notifications, config, deps, etc.
#
# ---------------------------------
# Дополнительные рекомендации
# ---------------------------------
# • Первая строка ≤ 72 символов, без точки и заглавных букв.
# • Если нужно подробное описание, добавь блок ниже.
# • Не пиши в прошедшем времени — коммит описывает действие.
# • На английском можно писать type/scope, а описание — по-русски.
#
# ---------------------------------
# Пример полного коммита
# ---------------------------------
# feat(reports): добавлена выгрузка отчётов в Excel
#
# Добавлен endpoint /reports/export
# Поддержаны форматы CSV и XLSX
# Добавлен worker для фоновой генерации файлов
#
# Добавить путь до шаблона:
# git config commit.template docs/.gitmessage.txt
# Удалить настройки шаблона:
# git config --unset commit.template
#
# ---------------------------------
# Конец шаблона
